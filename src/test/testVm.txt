vars
  i, x, res: int;

int function fact(m: int)
{
  if (m > 1) {
    return (m * fact(m - 1));
  } else {
    return(1);
  }
}

int function fib(n: int)
{
  if (n < 2) {
    return (n);
  } else {
    return (fib(n - 1) + fib(n - 2));
  }
}

main ()
{
  println("Resultado fact");
  println(fact(4));
  println("Resulatdo fib");
  println(fib(12));
}

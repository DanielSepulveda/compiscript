## Fibonacci recursivo

```
Program test;

int function fib(n: int){
  if(n < 2) {
    return(n);
  } else {
    return (fib(n - 1) + fib(n - 2));
  }
}

main() {
  println(fib(12));
}
```

## Factorial iterativo

```
Program test;
vars
  i, ans: int;

main() {
  ans = 1;

  for (i = 1) until (7) {
    ans = ans * i;
    i = i + 1;
  }

  print(ans);
}
```

## Bubble sort

```
Program bubbleSort;
vars
  v[4], i, j, tmp: int;

main() {
  read(v[0], v[1], v[2], v[3]);

  i = 0;
  j = 0;
  while(i != 4) {
    j = 0;
    while(j != 3) {
      if(v[j] > v[j+1]) {
        tmp = v[j];
        v[j] = v[j+1];
        v[j + 1] = tmp;
      }
      j = j + 1;
    }
    i = i + 1;
  }

  println(v[0], v[1], v[2], v[3]);
}
```

## Busqueda de valor en vector

```
Program test;
vars
  v[5], i, n: int;

int function find(n: int) {
  i = 0;
  while (i < 5) {
    if (v[i] == n) {
      return (1);
    }
    i = i + 1;
  }
  return (0);
}

main(){
  println("Ingresa valores");
  i = 0;
  while (i < 5) {
    println("Ingresa valor #", i + 1);
    read(v[i]);
    i = i + 1;
  }

  println("Ingresa valor a buscar");
  read(n);
  if (find(n)) {
    println("Valor encontrado");
  } else {
    println("Valor no encontrado");
  }
}
```
